Table identifiers
* identifier_id serial
* document_namespace_id integer
* id_string varchar( 255 )
  document_id integer
  package_id integer
  package_file_id integer
Indexes
Pkidentifiers_pkey ON identifier_id
Uuc_identifier_document_namespace_id_name ON document_namespace_id, id_string
Uuc_identifier_namespace_document_id ON document_namespace_id, document_id
Uuc_identifier_namespace_package_file_id ON document_namespace_id, package_file_id
Uuc_identifier_namespace_package_id ON document_namespace_id, package_id
Foreign Keys
 identifiers_document_namespace_id_fkey ( document_namespace_id ) ref document_namespaces (document_namespace_id)
 identifiers_document_id_fkey ( document_id ) ref documents (document_id)
 identifiers_package_id_fkey ( package_id ) ref packages (package_id)
 identifiers_package_file_id_fkey ( package_file_id ) ref packages_files (package_file_id)
Constraints
  ck_identifier_exactly_one ((((document_id IS NOT NULL))::integer + ((package_id IS NOT NULL))::integer) + ((package_file_id IS NOT NULL))::integer) = 1
